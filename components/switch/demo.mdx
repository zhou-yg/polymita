import { useState } from 'react'
import * as Module from './index.tsx'
import { RenderToReactWithWrap } from '../../shared/render'
export const Component = RenderToReactWithWrap(Module)

export function SwitchBox () {
  const [val, setVal] = useState(true)
  return (
    <div style={{ margin: '10px', color: '#999' }}>
      当前值: {String(val)} 
      <br/>
      <Component
        value={val} 
        onChange={v => setVal(v)} 
      />
    </div>
  )
}

export function SwitchBox2 () {
  const [val, setVal] = useState(true)
  return (
    <div style={{ margin: '10px', color: '#999' }}>
      当前值: {String(val)} 
      <br/>
      <Component
        checkedContent="OK" 
        uncheckedContent="Not Good"
        value={val} 
        onChange={v => setVal(v)} 
      />
    </div>
  )
}

# Switch 开关
